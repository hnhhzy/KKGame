cmd: [0x00, 0x1E]
desc: 时间指示及屏幕亮度蒙层
data_format:
    0x00: environment

    environment < 0x10:
    // 屏幕亮度蒙层控制(天黑)
    0x00: 白天
    0x01: 黄昏
    0x02: 夜间
    0x03: 黎明

    environment >= 0x14 && environment <= 0x2B
    environment-0x14: 当前是当天第几个小时
    0x14: 0点
    0x2B: 23点

    
    0x00, 0x01, 0x00, 0x1E, 0x00, 0x00, 0x00, 0x01,
    0x1D
